\textbf{Abstract.} This paper has two goals. First, it identifies the object that classical microprice estimates after the event-time model is reduced to price-move time. For a finite top-of-book Markov model with no-move kernel $Q$, move kernel $T$, event-time one-step drift
\[
r(x):=\E[p_{n+1}-p_n\mid X_n=x],
\]
and fundamental matrix $N=(I-Q)^{-1}$, define
\[
G_1:=Nr,
\qquad
B:=NT.
\]
The $n$-move predictor is then
\[
G^{(n)}=\sum_{j=0}^{n-1}B^jG_1.
\]
Second, it explains why the long-run correction can fail at short horizons even when the move-time model is well specified. If the move-time chain is ergodic with stationary distribution $\pi$, mean move drift $\mu=\pi^\top G_1$, and projector $\PiB:=\1\pi^\top$, then
\[
G^{(n)}=n\mu\,\1+g-B^n g,
\qquad
(I-B)g=G_1-\mu\1,
\qquad
\pi^\top g=0,
\]
and the long-run move-time microprice level is
\[
G^*:=(I-B+\PiB)^{-1}G_1=g+\mu\1.
\]
Thus classical microprice is a centered Poisson corrector on the move-time chain, while the transient term $-B^n g$ governs the horizon effect. We also study the move-time discounted resolvent $(I-\beta B)^{-1}G_1$, where $\beta$ discounts across price moves after the reduction; this is distinct from the event-time discount $\gamma$ used in the companion note \emph{Microprice, Bellman Values, and Semi-Markov Reinforcement Learning}.

Empirically, this decomposition predicts a horizon effect: at short horizons the transient term can still dominate, so $G_1$ or even the raw mid can outperform $G^*$; at longer horizons the centered correction wins when the move-time martingale diagnostic remains small. The public BAC/CVX benchmarks are mixed, WSELOB shows the clearest crossover, and a one-day IEX panel isolates the residual-drift failure mode.
\end{tcolorbox}

\section{Introduction}

Stoikov's microprice \cite{Stoikov2018} propagates expected future price changes through an absorbing top-of-book Markov chain. This paper is the structural half of a two-note sequence. It identifies the long-run object computed by the classical construction and explains why that object need not be the best short-horizon forecast.

Throughout both notes we use the same dictionary:
\[
r(x):=\E[p_{n+1}-p_n\mid X_n=x],
\qquad
N:=(I-Q)^{-1},
\qquad
G_1:=Nr,
\qquad
B:=NT.
\]
If the move-time chain $B$ is ergodic with stationary distribution $\pi$, we write
\[
\mu:=\pi^\top G_1,
\qquad
\PiB:=\1\pi^\top,
\qquad
(I-B)g=G_1-\mu\1,
\qquad
\pi^\top g=0,
\]
and
\[
G^*:=(I-B+\PiB)^{-1}G_1=g+\mu\1.
\]
Here $g$ is the centered move-time relative value and $G^*$ the uncentered long-run move-time microprice level; in centered regimes, $G^*=g$.

Paper~I analyzes the move-time Poisson decomposition, the martingale correction, and the horizon effect. Paper~II keeps the same objects $(r,N,G_1,B,g,G^*)$ and adds exact event-time Bellman targets together with move-time TD/SMDP learning rules. Empirically, the public BAC/CVX benchmarks are mixed, WSELOB gives the clearest horizon crossover, and the one-day IEX panel isolates the residual-drift failure mode.

\section{From event time to move time}

Let $X_n$ be a finite-state Markov approximation of the top of book at event time, and let $p_n$ be the mid-price. Decompose the transition kernel according to whether the mid-price moves:
\[
P=Q+T,
\]
where $Q_{xy}$ is the probability of moving from state $x$ to state $y$ without changing the mid-price and $T_{xy}$ is the probability of moving from $x$ to $y$ with a price change. Define the event-time one-step drift
\[
r(x):=\E[p_{n+1}-p_n\mid X_n=x].
\]

Assume $\rho(Q)<1$, so the no-move subchain is transient and
\[
N:=(I-Q)^{-1}=\sum_{k\ge 0}Q^k
\]
exists. Set
\[
G_1:=Nr,
\qquad
B:=NT.
\]

\begin{proposition}[Embedded move-time chain]\label{prop:embedded}
Let
\[
\tau_0:=0,
\qquad
\tau_{k+1}:=\inf\{n>\tau_k:p_n\neq p_{n-1}\},
\qquad
Y_k:=X_{\tau_k}.
\]
Then $Y_k$ is a Markov chain with transition kernel $B$. Moreover,
\[
G_1(x)=\E_x[p_{\tau_1}-p_{\tau_0}],
\]
so $G_1$ is the expected price change at the next move conditional on the current state.
\end{proposition}

Thus the event-time recursion collapses to move time: $B$ is the price-move kernel and $G_1$ the one-move drift. The companion note uses the discounted counterparts
\[
G_{1,\gamma}:=(I-\gamma Q)^{-1}r,
\qquad
B_\gamma:=\gamma(I-\gamma Q)^{-1}T,
\]
which discount each no-move excursion before it is collapsed. In this paper, $\beta$ is reserved for the move-time resolvent $(I-\beta B)^{-1}G_1$, which discounts only across already-collapsed price moves.

If $\pi$ denotes the stationary distribution of $B$, define
\[
\mu:=\pi^\top G_1,
\qquad
\PiB:=\1\pi^\top.
\]

\begin{remark}[Centering]
When $B$ is ergodic, $B^n\to \PiB$. Hence Stoikov's convergence condition $B^nG_1\to 0$ is equivalent to
\[
\pi^\top G_1=0.
\]
Only the invariant constant mode can obstruct convergence.
\end{remark}

\section{The centered Poisson corrector}

Define the $n$-move price adjustment by
\[
G^{(n)}:=\sum_{j=0}^{n-1}B^jG_1.
\]
Equivalently,
\[
\E[p_{\tau_n}-p_{\tau_0}\mid Y_0=x]=G^{(n)}(x).
\]

\begin{proposition}[Finite-horizon decomposition]\label{prop:finite}
Assume that $B$ is ergodic with stationary distribution $\pi$, and set
\[
h:=G_1-\mu\1.
\]
Let $g$ be the unique centered solution of
\[
(I-B)g=h,
\qquad
\pi^\top g=0.
\]
Then, for every $n\ge 1$,
\[
G^{(n)}=n\mu\,\1+g-B^n g.
\]
If
\[
G^*:=(I-B+\PiB)^{-1}G_1,
\]
then
\[
G^*=g+\mu\1.
\]
In particular, if $\mu=0$, then $G^*=g$ and
\[
G^{(n)}\to G^*.
\]
\end{proposition}

Thus the predictor splits into linear drift, a bounded state correction, and the transient remainder $-B^n g$. The empirical horizon effect is entirely that last term.

\begin{corollary}[Move-time martingale correction]\label{cor:mart}
Under the assumptions of Proposition~\ref{prop:finite},
\[
M_k:=p_{\tau_k}-k\mu+g(Y_k)
\]
is a martingale with respect to the move-time filtration. When $\mu=0$, the corrected price
\[
p_{\tau_k}+G^*(Y_k)
\]
is a move-time martingale.
\end{corollary}

\begin{proposition}[Move-time discounted regularization]\label{prop:disc}
For $\beta\in(0,1)$ define
\[
g_\beta:=(I-\beta B)^{-1}G_1.
\]
Then
\[
g_\beta=\frac{\mu}{1-\beta}\,\1+(I-\beta B)^{-1}(G_1-\mu\1).
\]
Hence the only singular term as $\beta\uparrow 1$ is the constant mode. If $\mu=0$ and, for some $\rho\in(0,1)$,
\[
\norm{B^kG_1}_\infty\le C_\rho\rho^k\norm{G_1}_\infty
\qquad
\text{for all }k\ge 0,
\]
then
\[
\norm{g_\beta-G^*}_\infty
\le
(1-\beta)\frac{C_\rho\rho}{(1-\rho)^2}\norm{G_1}_\infty.
\]
\end{proposition}

The resolvent is therefore best viewed as a regularized proxy for $G^*$: nearly lossless when $\mu\approx 0$, but dominated by the constant mode when centering fails. Unlike $V_\gamma=(I-\gamma P)^{-1}r$ in Paper~II, it discounts only after the move-time reduction.

\section{Empirical results}

\subsection{Evaluation protocol}

We evaluate predictors against grouped state-conditioned future-price curves rather than pointwise forecast errors. For a horizon $h$, let $c$ index a spread/imbalance bucket. Write $\bar \Delta_h(c)$ for the average future mid-price change over all rows in bucket $c$, and write $\bar f(c)$ for the average predictor value in the same bucket. The reported score is
\[
\mathrm{RMSE}_h(f):=
\left(
\frac{1}{|C_h|}
\sum_{c\in C_h}\bigl(\bar \Delta_h(c)-\bar f(c)\bigr)^2
\right)^{1/2}.
\]
The horizons are fixed clock/event-time horizons rather than exact move counts, so the empirical question is qualitative: does the long-run correction become more useful as the horizon lengthens? Throughout this paper, ``Disc(.99)'' means the move-time resolvent $(I-0.99\,B)^{-1}G_1$; the companion note reserves $V_{0.99}$ for event-time discounting on $P$.

To connect the data back to Corollary~\ref{cor:mart}, we also report the move-time martingale diagnostic. Writing
\[
\Delta P_k:=p_{\tau_{k+1}}-p_{\tau_k},
\qquad
\widehat \xi_k:=\Delta P_k-\widehat\mu+G^*(Y_{k+1})-G^*(Y_k),
\qquad
\widehat t_M:=\left|\frac{\overline{\widehat \xi}}{s_{\widehat \xi}/\sqrt{n}}\right|.
\]
Small values of $\widehat t_M$ indicate that the estimated correction is close to a move-time martingale.

\subsection{WSELOB: the horizon crossover}

WSELOB gives the clearest horizon crossover. At 10 seconds, the raw mid is best on PEKAO, PKOBP, and PZU, while $G_1$ is best on KGHM and PKNORLEN; $G^*$ can still be between $6\%$ and $83\%$ worse than the raw mid, consistent with a material transient. By 60 seconds the ordering reverses: $G^*$ is best on four names and Disc(.99) on the fifth, with grouped-RMSE reductions of $20\%$ to $56\%$ relative to the raw mid. The weighted mid is not competitive, and $\widehat t_M<2$ for every name.

\subsection{IEX: centering as a diagnostic}

The IEX venue-local panel is deliberately harder: a single venue, a short sample, and a coarse state description leave more room for residual drift. BAC remains near-centered with $\widehat t_M=0.16$ and $G^*$ modestly improves on the raw and weighted mid. SPY and GLD do not: $\widehat t_M=3.09$ and $2.24$, and the discounted surrogate no longer shadows the centered correction. This is the failure mode of Proposition~\ref{prop:disc}: when the constant mode is not well centered, discounted extrapolation is driven by residual drift rather than by the bounded state correction.

\section{Conclusion}

Sampling at move times turns microprice into a finite-state Poisson problem. The event-time recursion collapses to a move-time kernel $B$ and a one-move drift $G_1$, and the $n$-move predictor decomposes as
\[
G^{(n)}=n\mu\,\1+g-B^n g.
\]
This identity isolates the martingale correction and the horizon effect.

Empirically, the correction works best when the horizon is long enough for $B^n g$ to decay and the move-time martingale diagnostic remains small. WSELOB is the clearest positive case, BAC/CVX are useful public calibrations, and the IEX panel shows the residual-drift failure mode. Paper~II starts from the same primitives $(r,N,G_1,B,g,G^*)$ and adds exact event-time Bellman objectives together with move-time TD/SMDP learning rules.

\section{Proofs}

\begin{proof}[Proof of Proposition~\ref{prop:embedded}]
The first price move occurs after $n$ no-move steps and one move step with kernel $Q^nT$. Summing over $n\ge 0$ gives
\[
\Prob_x(Y_1=y)=\sum_{n\ge 0}(Q^nT)_{xy}
=((I-Q)^{-1}T)_{xy}
=B_{xy}.
\]
Thus $B$ is the move-time transition kernel. For the reward identity,
\[
\E_x[p_{\tau_1}-p_{\tau_0}]
=
\sum_{n\ge 0}(Q^nr)_x
=((I-Q)^{-1}r)_x
=G_1(x).
\]
The Markov property of $(Y_k)$ follows from the strong Markov property at the stopping times $(\tau_k)$.
\end{proof}

\begin{proof}[Proof of Proposition~\ref{prop:finite}]
Write $G_1=\mu\1+h$ with $\pi^\top h=0$. Since $h=(I-B)g$,
\[
G^{(n)}
=
\sum_{j=0}^{n-1}B^jG_1
=
n\mu\,\1+\sum_{j=0}^{n-1}B^j(I-B)g
=
n\mu\,\1+g-B^n g.
\]
Also,
\[
(I-B+\PiB)(g+\mu\1)=(I-B)g+\PiB g+\mu\PiB\1=h+\mu\1=G_1,
\]
so $G^*=g+\mu\1$. If $\mu=0$, then $G^*=g$ and ergodicity gives $B^n g\to \PiB g=\1\pi^\top g=0$, hence $G^{(n)}\to G^*$.
\end{proof}

\begin{proof}[Proof of Corollary~\ref{cor:mart}]
By Proposition~\ref{prop:finite},
\[
(I-B)g=G_1-\mu\1.
\]
Hence
\[
\E[p_{\tau_{k+1}}-p_{\tau_k}\mid Y_k]=G_1(Y_k),
\]
while
\[
\E[g(Y_{k+1})-g(Y_k)\mid Y_k]=(Bg-g)(Y_k)=-(G_1(Y_k)-\mu).
\]
Adding the two conditional expectations yields
\[
\E[M_{k+1}-M_k\mid Y_k]=0.
\]
\end{proof}

\begin{proof}[Proof of Proposition~\ref{prop:disc}]
The Neumann series gives
\[
g_\beta=\sum_{k\ge 0}\beta^k B^kG_1.
\]
With $G_1=\mu\1+h$ and $B^k\1=\1$,
\[
g_\beta=\sum_{k\ge 0}\beta^k(\mu\1+B^k h)=\frac{\mu}{1-\beta}\,\1+(I-\beta B)^{-1}h.
\]
If $\mu=0$, then $h=G_1$ and
\[
G^*-g_\beta=\sum_{k\ge 0}(1-\beta^k)B^kG_1.
\]
The $k=0$ term vanishes, and for $k\ge 1$ we have $1-\beta^k\le k(1-\beta)$. Hence
\[
\norm{G^*-g_\beta}_\infty
\le
(1-\beta)C_\rho\norm{G_1}_\infty\sum_{k\ge 1}k\rho^k
=
(1-\beta)\frac{C_\rho\rho}{(1-\rho)^2}\norm{G_1}_\infty.
\]
\end{proof}
