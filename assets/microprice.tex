\textbf{Abstract.} Stoikov's microprice is most transparent on the embedded chain observed only at mid-price move times. For a finite top-of-book Markov model with no-move kernel $Q$, move kernel $T$, and fundamental matrix $N=(I-Q)^{-1}$, define the one-move drift $G_1=N\bar r$ and the move-time kernel $B=NT$. The $n$-move predictor is then
\[
G^{(n)}=\sum_{j=0}^{n-1}B^jG_1.
\]
If the move-time chain is ergodic with stationary distribution $\pi$ and stationary drift $\mu=\pi^\top G_1$, then
\[
G^{(n)}=n\mu\,\1+g-B^n g,
\qquad
(I-B)g=G_1-\mu\1,
\qquad
\pi^\top g=0.
\]
The long-run state adjustment is therefore the centered Poisson solution of the move-time chain, and
\[
p_{\tau_k}-k\mu+g(Y_k)
\]
is a martingale at move times. We also study the discounted resolvent $(I-\alpha B)^{-1}G_1$, which regularizes the constant mode and converges to the centered correction in centered regimes.

Empirically, this decomposition predicts a horizon effect: at short horizons the transient term can still dominate, so $G_1$ or even the raw mid can outperform the long-run correction; at longer horizons the centered correction wins when the move-time martingale diagnostic remains small. Public BAC/CVX benchmarks, a five-name WSELOB panel, and a one-day IEX stress test all support that pattern. On WSELOB, the raw mid or $G_1$ is best at 10 seconds, whereas $G^*$ or Disc(.99) is best on all five names by 60 seconds.
\end{tcolorbox}

\section{Introduction}

Stoikov's microprice \cite{Stoikov2018} estimates fair value from top-of-book state by propagating expected future price changes through an absorbing Markov chain. The construction is useful, but its operator structure becomes much clearer after sampling only at price-move times. In that representation the problem reduces to a finite-state Poisson equation on the move-time chain \cite{KemenySnell1976,Meyer1975}.

This paper isolates that move-time representation and uses it to sharpen both the theory and the empirical interpretation of microprice estimates. The event-time model collapses to a move-time kernel $B$ and a one-move drift $G_1$. The $n$-move predictor then admits the exact decomposition
\[
G^{(n)}=n\mu\,\1+g-B^n g,
\]
which separates a linear drift, a bounded state correction, and a transient remainder. The centered microprice is the bounded component, while the transient term explains why the long-run correction need not be best at the shortest horizons.

The empirical section evaluates that prediction on three datasets. The public BAC/CVX benchmarks remain useful calibration checks, but they do not show uniform dominance: on BAC the weighted mid is still best under the grouped conditional-curve metric used here. The clearest positive evidence comes from WSELOB, where short- and long-horizon regimes separate cleanly. A one-day IEX venue-local panel then shows the complementary boundary case in which the centering diagnostic fails and discounted extrapolation becomes unreliable.

\section{From event time to move time}

Let $X_n$ be a finite-state Markov approximation of the top of book at event time, and let $p_n$ be the mid-price. Decompose the transition kernel according to whether the mid-price moves:
\[
P=Q+T,
\]
where $Q_{xy}$ is the probability of moving from state $x$ to state $y$ without changing the mid-price and $T_{xy}$ is the probability of moving from $x$ to $y$ with a price change. Define the event-time one-step drift
\[
\bar r_x:=\E[p_{n+1}-p_n\mid X_n=x].
\]

Assume $\rho(Q)<1$, so the no-move subchain is transient and
\[
N:=(I-Q)^{-1}=\sum_{k\ge 0}Q^k
\]
exists. Set
\[
G_1:=N\bar r,
\qquad
B:=NT.
\]

\begin{proposition}[Embedded move-time chain]\label{prop:embedded}
Let
\[
\tau_0:=0,
\qquad
\tau_{k+1}:=\inf\{n>\tau_k:p_n\neq p_{n-1}\},
\qquad
Y_k:=X_{\tau_k}.
\]
Then $Y_k$ is a Markov chain with transition kernel $B$. Moreover,
\[
G_1(x)=\E_x[p_{\tau_1}-p_{\tau_0}],
\]
so $G_1$ is the expected price change at the next move conditional on the current state.
\end{proposition}

The absorbing-chain recursion therefore reduces to a move-time Markov system: $B$ controls transitions from one price move to the next, and $G_1$ is the corresponding one-move price drift. If $\pi$ denotes the stationary distribution of $B$, define
\[
\mu:=\pi^\top G_1,
\qquad
\Piop:=\1\pi^\top.
\]

\begin{remark}[Centering]
When $B$ is ergodic, $B^n\to \Piop$. Hence Stoikov's convergence condition $B^nG_1\to 0$ is equivalent to
\[
\pi^\top G_1=0.
\]
Only the invariant constant mode can obstruct convergence.
\end{remark}

\section{The centered Poisson corrector}

Define the $n$-move price adjustment by
\[
G^{(n)}:=\sum_{j=0}^{n-1}B^jG_1.
\]
Equivalently,
\[
\E[p_{\tau_n}-p_{\tau_0}\mid Y_0=x]=G^{(n)}(x).
\]

\begin{proposition}[Finite-horizon decomposition]\label{prop:finite}
Assume that $B$ is ergodic with stationary distribution $\pi$, and set
\[
h:=G_1-\mu\1.
\]
Let $g$ be the unique centered solution of
\[
(I-B)g=h,
\qquad
\pi^\top g=0.
\]
Then, for every $n\ge 1$,
\[
G^{(n)}=n\mu\,\1+g-B^n g.
\]
If
\[
G^*:=(I-B+\Piop)^{-1}G_1,
\]
then
\[
G^*=g+\mu\1.
\]
In particular, if $\mu=0$, then $G^*=g$ and
\[
G^{(n)}\to G^*.
\]
\end{proposition}

The predictor therefore splits into three pieces: a linear drift term $n\mu\1$, a bounded state correction $g$, and a transient remainder $-B^n g$. The centered microprice is the bounded component. The empirical horizon effect comes from the transient term: at short horizons it can still dominate the limiting correction.

\begin{corollary}[Move-time martingale correction]\label{cor:mart}
Under the assumptions of Proposition~\ref{prop:finite},
\[
M_k:=p_{\tau_k}-k\mu+g(Y_k)
\]
is a martingale with respect to the move-time filtration. When $\mu=0$, the corrected price
\[
p_{\tau_k}+G^*(Y_k)
\]
is a move-time martingale.
\end{corollary}

\begin{proposition}[Discounted regularization]\label{prop:disc}
For $\alpha\in(0,1)$ define
\[
g_\alpha:=(I-\alpha B)^{-1}G_1.
\]
Then
\[
g_\alpha=\frac{\mu}{1-\alpha}\,\1+(I-\alpha B)^{-1}(G_1-\mu\1).
\]
Hence the only singular term as $\alpha\uparrow 1$ is the constant mode. If $\mu=0$ and, for some $\rho\in(0,1)$,
\[
\norm{B^kG_1}_\infty\le C_\rho\rho^k\norm{G_1}_\infty
\qquad
\text{for all }k\ge 0,
\]
then
\[
\norm{g_\alpha-G^*}_\infty
\le
(1-\alpha)\frac{C_\rho\rho}{(1-\rho)^2}\norm{G_1}_\infty.
\]
\end{proposition}

The resolvent is always well posed, even when sample centering is imperfect. In centered regimes it is nearly lossless; in drifting or weakly centered regimes its constant mode can dominate. It is therefore best viewed as a regularized proxy and a useful stress test, not as an automatic substitute for the centered correction.

\section{Empirical results}

\subsection{Evaluation protocol}

We evaluate predictors against grouped state-conditioned future-price curves rather than pointwise forecast errors. For a horizon $h$, let $c$ index a spread/imbalance bucket. Write $\bar \Delta_h(c)$ for the average future mid-price change over all rows in bucket $c$, and write $\bar f(c)$ for the average predictor value in the same bucket. The reported score is
\[
\mathrm{RMSE}_h(f):=
\left(
\frac{1}{|C_h|}
\sum_{c\in C_h}\bigl(\bar \Delta_h(c)-\bar f(c)\bigr)^2
\right)^{1/2}.
\]
The horizons are fixed clock/event-time horizons rather than exact move counts, so the empirical question is qualitative: does the long-run correction become more useful as the horizon lengthens?

To connect the data back to Corollary~\ref{cor:mart}, we also report the move-time martingale diagnostic
\[
\widehat \xi_k:=\Delta p_k-\widehat\mu+G^*(Y_{k+1})-G^*(Y_k),
\qquad
\widehat t_M:=\left|\frac{\overline{\widehat \xi}}{s_{\widehat \xi}/\sqrt{n}}\right|.
\]
Small values of $\widehat t_M$ indicate that the estimated correction is close to a move-time martingale.

\subsection{Public BAC/CVX benchmarks}

Table~\ref{tab:public} and Figure~\ref{fig:public} summarize the public BAC/CVX benchmarks. On BAC, $G^*$ reduces grouped RMSE by about $51\%$ at 60 seconds and $76\%$ at 180 seconds relative to the raw mid, but the weighted mid remains best on this metric. On CVX, by contrast, $G^*$ reduces grouped RMSE by roughly $62\%$ and decisively beats the weighted mid. In all three benchmark cases, $\widehat t_M<2$.

\subsection{WSELOB: the horizon crossover}

WSELOB gives the clearest horizon crossover. At 10 seconds, the raw mid is best on PEKAO, PKOBP, and PZU, while $G_1$ is best on KGHM and PKNORLEN. On those same names, $G^*$ is still between $6\%$ and $83\%$ worse than the raw mid, which is exactly what one expects when the transient term $-B^n g$ remains material.

By 60 seconds the ordering reverses. $G^*$ is best on four names and Disc(.99) is best on the fifth. Relative to the raw mid, $G^*$ reduces grouped RMSE by between $20\%$ and $56\%$ on all five names. The weighted mid is not competitive: it is between $4.7\times$ and $11.0\times$ the raw-mid error at 10 seconds, and between $2.1\times$ and $6.6\times$ the $G^*$ error at 60 seconds. The move-time diagnostic remains comfortably in the centered regime, with $\widehat t_M<2$ for every name.

\subsection{IEX: centering as a diagnostic}

The IEX venue-local panel is deliberately harder: it uses a single venue, a short sample, and a coarse state description, so residual drift is more likely. BAC remains near-centered with $\widehat t_M=0.16$, and $G^*$ is modestly better than the raw or weighted mid at both horizons. SPY and GLD look different. Their diagnostics are $\widehat t_M=3.09$ and $2.24$, respectively, and the discounted surrogate no longer shadows the centered correction. At 10 seconds, Disc(.99) is about $3.16\times$ the $G^*$ error on GLD and $9.50\times$ the $G^*$ error on SPY. This is exactly the failure mode predicted by Proposition~\ref{prop:disc}: once the constant mode is not well centered, discounted extrapolation can be dominated by residual drift rather than by the bounded state correction.

\section{Conclusion}

Sampling at move times turns the microprice construction into a finite-state Poisson problem. The event-time recursion collapses to a move-time kernel $B$ and a one-move drift $G_1$, while the $n$-move predictor admits the exact decomposition
\[
G^{(n)}=n\mu\,\1+g-B^n g.
\]
This representation clarifies both the martingale correction and the empirical horizon dependence of the estimator.

Across datasets, the centered correction is most useful when two conditions hold at once: the horizon is long enough for the transient term to decay, and the move-time martingale diagnostic remains small. WSELOB is the clearest positive case. BAC/CVX confirm the public benchmark, with the important qualification that BAC still favors the weighted mid under the grouped metric used here. The IEX panel shows the complementary failure mode: when centering weakens, discounted regularization amplifies the wrong mode rather than recovering the bounded state correction.

\section{Proofs}

\begin{proof}[Proof of Proposition~\ref{prop:embedded}]
The first price move occurs after $n$ no-move steps and one move step with kernel $Q^nT$. Summing over $n\ge 0$ gives
\[
\Prob_x(Y_1=y)=\sum_{n\ge 0}(Q^nT)_{xy}
=((I-Q)^{-1}T)_{xy}
=B_{xy}.
\]
Thus $B$ is the move-time transition kernel. For the reward identity,
\[
\E_x[p_{\tau_1}-p_{\tau_0}]
=
\sum_{n\ge 0}(Q^n\bar r)_x
=((I-Q)^{-1}\bar r)_x
=G_1(x).
\]
The Markov property of $(Y_k)$ follows from the strong Markov property at the stopping times $(\tau_k)$.
\end{proof}

\begin{proof}[Proof of Proposition~\ref{prop:finite}]
Write $G_1=\mu\1+h$ with $\pi^\top h=0$. Since $(I-B)g=h$,
\[
\sum_{j=0}^{n-1}B^j h
=
\sum_{j=0}^{n-1}B^j(I-B)g
=
\sum_{j=0}^{n-1}(B^jg-B^{j+1}g)
=
g-B^n g.
\]
Therefore
\[
G^{(n)}
=
\sum_{j=0}^{n-1}B^jG_1
=
n\mu\,\1+g-B^n g.
\]
Next,
\[
(I-B+\Piop)(g+\mu\1)
=
(I-B)g+\Piop g+\mu\Piop\1
=
h+0+\mu\1
=
G_1,
\]
so $G^*=g+\mu\1$. If $\mu=0$, then $G^*=g$ and ergodicity gives
\[
B^ng\to \Piop g=\1\pi^\top g=0,
\]
hence $G^{(n)}\to G^*$. Uniqueness of $g$ under $\pi^\top g=0$ follows because the nullspace of $I-B$ consists of constants.
\end{proof}

\begin{proof}[Proof of Corollary~\ref{cor:mart}]
By Proposition~\ref{prop:finite},
\[
(I-B)g=G_1-\mu\1.
\]
Hence
\[
\E[p_{\tau_{k+1}}-p_{\tau_k}\mid Y_k]=G_1(Y_k),
\]
while
\[
\E[g(Y_{k+1})-g(Y_k)\mid Y_k]=(Bg-g)(Y_k)=-(G_1(Y_k)-\mu).
\]
Adding the two conditional expectations yields
\[
\E[M_{k+1}-M_k\mid Y_k]=0.
\]
\end{proof}

\begin{proof}[Proof of Proposition~\ref{prop:disc}]
Because $\|\alpha B\|_\infty\le \alpha<1$, the Neumann series converges and
\[
g_\alpha=\sum_{k\ge 0}\alpha^k B^kG_1.
\]
Using $G_1=\mu\1+h$ and $B^k\1=\1$,
\[
g_\alpha
=
\sum_{k\ge 0}\alpha^k(\mu\1+B^kh)
=
\frac{\mu}{1-\alpha}\,\1+(I-\alpha B)^{-1}h.
\]
If $\mu=0$, then $h=G_1$ and
\[
G^*-g_\alpha
=
\sum_{k\ge 0}(1-\alpha^k)B^kG_1.
\]
The $k=0$ term vanishes. For $k\ge 1$, $1-\alpha^k\le k(1-\alpha)$, so
\[
\norm{G^*-g_\alpha}_\infty
\le
(1-\alpha)\sum_{k\ge 1}k\,\norm{B^kG_1}_\infty
\le
(1-\alpha)C_\rho\norm{G_1}_\infty\sum_{k\ge 1}k\rho^k.
\]
Since $\sum_{k\ge 1}k\rho^k=\rho/(1-\rho)^2$, the stated bound follows.
\end{proof}
